/etc/config/wireless

config wifi-device 'radio0'
        option type 'mac80211'
        option hwmode '11g'
        option path 'platform/10300000.wmac'
        option htmode 'HT20'
        option disabled '0'
        option country '00'

     option legacy_rates '1'
        option channel '6'

config wifi-iface
        option network 'wifirep'
        option ssid 'Livebox-442c'

option encryption 'psk2'
        option device 'radio0'
        option mode 'sta'
        option bssid '$BSSID'
        option key ' '
